# 비트연산을 이용한 부분집합

```swift
let numbers = [1,2,3,4]

// count = 4
// 1 << 4 == 0b10000
// 0001 ~ 1111 까지
for i in 0..<(1 << numbers.count) {
    print(String(i, radix: 2))

    var sub: [Int] = []

    // 각각의 원소가 있는지 확인
    // (1 << j) == 1(0001), 2(0010), 4(0100), 8(1000) &연산
    // 1 = 0번째의 원소가 있는지 확인. i 와 & 연산하여
    for j in 0..<numbers.count {

        // 0 이라면 (1 << j) 번째 원소가 i에 존재하지 않는다.
        // i.g. 9(1001) & 2(0010) = 0
        // 9(1001) & 1(1) = 1
        // 9(1001) & 4(1000) = 4
        if i & (1 << j) != 0 {
            sub.append(numbers[j])
        }

    }

    print(sub)
    print()
}
```
