# 순열, 조합

```swift

// n개 중 k개
func getPerm(n: Int, k: Int) -> [[Int]] {
    var result: [[Int]] = []
    var visited = Array(repeating: false, count: n)
    
    func perm(_ output: [Int] = []) {
        if output.count == k {
            result.append(output)
            return
        }
        
        for i in 0..<n {
            if visited[i] == false {
                visited[i] = true
                perm(output + [i])
                visited[i] = false
            }
        }
    }
    perm()
    
    return result
}


func getComb(n: Int, k: Int) -> [[Int]] {
    var result: [[Int]] = []
    
    func comb(_ index: Int = 0, _ output: [Int] = []) {
        if output.count == k {
            result.append(output)
            return
        }
        for i in index..<n {
            comb(i + 1, output + [i])
        }
    }
    comb()
    
    return result
}
```
